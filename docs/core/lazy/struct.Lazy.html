Expand description

在首次访问时初始化的值。

Examples

#![feature(once_cell)]

use std::lazy::Lazy;

let lazy: Lazy<i32> = Lazy::new(|| {
    println!("initializing");
    92
});
println!("ready");
println!("{}", *lazy);
println!("{}", *lazy);

// Prints:
//   准备初始化
//   92
//   92
