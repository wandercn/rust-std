Expand description

将参数转换为迭代器并压缩它们。

有关更多信息，请参见 Iterator::zip 的文档。

Examples

#![feature(iter_zip)]
use std::iter::zip;

let xs = [1, 2, 3];
let ys = [4, 5, 6];
for (x, y) in zip(&xs, &ys) {
    println!("x:{}, y:{}", x, y);
}

// 嵌套 zip 也可以:
let zs = [7, 8, 9];
for ((x, y), z) in zip(zip(&xs, &ys), &zs) {
    println!("x:{}, y:{}, z:{}", x, y, z);
}
