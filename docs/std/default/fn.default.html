pub fn default<T>() -> T where
    T: Default, 
🔬 This is a nightly-only experimental API. (default_free_fn #73014)
Expand description

根据 Default trait 返回类型的默认值。

从上下文推断要返回的类型。这等效于 Default::default()，但类型更短。

例如:

#![feature(default_free_fn)]

use std::default::default;

#[derive(Default)]
struct AppConfig {
    foo: FooConfig,
    bar: BarConfig,
}

#[derive(Default)]
struct FooConfig {
    foo: i32,
}

#[derive(Default)]
struct BarConfig {
    bar: f32,
    baz: u8,
}

fn main() {
    let options = AppConfig {
        foo: default(),
        bar: BarConfig {
            bar: 10.1,
            ..default()
        },
    };
}
