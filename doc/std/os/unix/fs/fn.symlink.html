pub fn symlink<P: AsRef<Path>, Q: AsRef<Path>>(
    original: P, 
    link: Q
) -> Result<()>
This is supported on Unix only.
Expand description

在文件系统上创建一个新的符号链接。

link 路径将是指向 original 路径的符号链接。

Examples

use std::os::unix::fs;

fn main() -> std::io::Result<()> {
    fs::symlink("a.txt", "b.txt")?;
    Ok(())
}
